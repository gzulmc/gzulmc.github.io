class Config:
    # MySQL配置
    MYSQL_HOST = 'localhost'
    MYSQL_PORT = 3306
    MYSQL_USER = 'root'
    MYSQL_PASSWORD = 'password'
    MYSQL_DB = 'rbac_system'

    # Redis配置
    REDIS_HOST = 'localhost'

    SECRET_KEY = 'your-secret-key-here'

# 数据库连接管理
class DatabaseManager:
    def __init__(self):
        self.mysql_config = {
            'host': Config.MYSQL_HOST,
            'port': Config.MYSQL_PORT,
            'user': Config.MYSQL_USER,
            'password': Config.MYSQL_PASSWORD,
            'database': Config.MYSQL_DB,
            'charset': 'utf8mb4',
            'autocommit': True
        }
        self.redis_client = redis.Redis(
            host=Config.REDIS_HOST,
            port=Config.REDIS_PORT,
            db=Config.REDIS_DB,
            decode_responses=True
        )

    def get_mysql_connection(self):
        return pymysql.connect(**self.mysql_config)

    def get_redis_client(self):
        return self.redis_client

# 初始化数据库
def init_database():
    """初始化数据库表结构"""
    db = DatabaseManager()
    conn = db.get_mysql_connection()
    cursor = conn.cursor()

    # 创建数据库
    cursor.execute(f"CREATE DATABASE IF NOT EXISTS {Config.MYSQL_DB}")
    cursor.execute(f"USE {Config.MYSQL_DB}")

    # 用户表
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS users (
            id INT AUTO_INCREMENT PRIMARY KEY,
            username VARCHAR(50) UNIQUE NOT NULL,
            password_hash VARCHAR(255) NOT NULL,
            email VARCHAR(100),
            status TINYINT DEFAULT 1 COMMENT '1-激活, 0-禁用',
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
        )
    ''')

    # 角色表
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS roles (
            id INT AUTO_INCREMENT PRIMARY KEY,
            name VARCHAR(50) UNIQUE NOT NULL,
            description TEXT,
            status TINYINT DEFAULT 1 COMMENT '1-激活, 0-禁用',
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
        )
    ''')

    # 权限表
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS permissions (
            id INT AUTO_INCREMENT PRIMARY KEY,
            name VARCHAR(50) UNIQUE NOT NULL,
            resource VARCHAR(100) NOT NULL COMMENT '资源名称',
            action VARCHAR(50) NOT NULL COMMENT '操作类型',
            description TEXT,
            status TINYINT DEFAULT 1 COMMENT '1-激活, 0-禁用',
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
        )
    ''')

    # 用户角色关联表
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS user_roles (
            id INT AUTO_INCREMENT PRIMARY KEY,
            user_id INT NOT NULL,
            role_id INT NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
            FOREIGN KEY (role_id) REFERENCES roles(id) ON DELETE CASCADE,
            UNIQUE KEY unique_user_role (user_id, role_id)
        )
    ''')

    # 角色权限关联表
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS role_permissions (
            id INT AUTO_INCREMENT PRIMARY KEY,
            role_id INT NOT NULL,
            permission_id INT NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (role_id) REFERENCES roles(id) ON DELETE CASCADE,
            FOREIGN KEY (permission_id) REFERENCES permissions(id) ON DELETE CASCADE,
            UNIQUE KEY unique_role_permission (role_id, permission_id)
        )
    ''')

    conn.close()
    print("数据库表创建成功！")

# Token管理器
class TokenManager:
    def __init__(self, db_manager: DatabaseManager):
        self.db_manager = db_manager
        self.redis_client = db_manager.get_redis_client()

    def generate_token(self, user_id: int) -> str:
        """生成token"""
        token = secrets.token_urlsafe(32)

        # 存储到Redis，设置过期时间
        token_data = {
            'user_id': user_id,
            'created_at': datetime.now().isoformat(),
            'expires_at': (datetime.now() + timedelta(hours=Config.TOKEN_EXPIRE_HOURS)).isoformat()
        }

        self.redis_client.setex(
            f"token:{token}",
            Config.TOKEN_EXPIRE_HOURS * 3600,
            json.dumps(token_data)
        )

        return token

    def verify_token(self, token: str) -> Optional[Dict]:
        """验证token"""
        token_data = self.redis_client.get(f"token:{token}")
        if not token_data:
            return None

        try:
            data = json.loads(token_data)
            expires_at = datetime.fromisoformat(data['expires_at'])

            if datetime.now() > expires_at:
                self.redis_client.delete(f"token:{token}")
                return None

            return data
        except:
            return None

    def revoke_token(self, token: str):
        """撤销token"""
        self.redis_client.delete(f"token:{token}")

# 权限管理器
class PermissionManager:
    def __init__(self, db_manager: DatabaseManager):
        self.db_manager = db_manager
        self.redis_client = db_manager.get_redis_client()

    def get_user_permissions(self, user_id: int) -> Set[str]:
        """获取用户权限"""
        cache_key = f"user_permissions:{user_id}"

        # 先从Redis缓存获取
        cached_permissions = self.redis_client.get(cache_key)
        if cached_permissions:
            return set(json.loads(cached_permissions))

        # 从数据库查询
        conn = self.db_manager.get_mysql_connection()
        cursor = conn.cursor()

        query = '''
            SELECT DISTINCT p.name, p.resource, p.action
            FROM permissions p
            JOIN role_permissions rp ON p.id = rp.permission_id
            JOIN roles r ON rp.role_id = r.id
            JOIN user_roles ur ON r.id = ur.role_id
            WHERE ur.user_id = %s AND p.status = 1 AND r.status = 1
        '''

        cursor.execute(query, (user_id,))
        results = cursor.fetchall()
        conn.close()

        permissions = set()
        for result in results:
            # 格式: resource:action
            permission = f"{result[1]}:{result[2]}"
            permissions.add(permission)

        # 缓存到Redis，5分钟过期
        self.redis_client.setex(cache_key, 300, json.dumps(list(permissions)))

        return permissions

    def check_permission(self, user_id: int, resource: str, action: str) -> bool:
        """检查用户是否有指定权限"""
        permissions = self.get_user_permissions(user_id)
        required_permission = f"{resource}:{action}"
        return required_permission in permissions

    def clear_user_permissions_cache(self, user_id: int):
        """清除用户权限缓存"""
        cache_key = f"user_permissions:{user_id}"
        self.redis_client.delete(cache_key)

# 用户管理器
class UserManager:
    def __init__(self, db_manager: DatabaseManager):
        self.db_manager = db_manager

    def create_user(self, username: str, password: str, email: str = None) -> Dict:
        """创建用户"""
        conn = self.db_manager.get_mysql_connection()
        cursor = conn.cursor()

        try:
            password_hash = generate_password_hash(password)

            cursor.execute('''
                INSERT INTO users (username, password_hash, email)
                VALUES (%s, %s, %s)
            ''', (username, password_hash, email))

            user_id = cursor.lastrowid
            conn.close()

            return {
                'id': user_id,
                'username': username,
                'email': email,
                'status': 1
            }
        except pymysql.IntegrityError:
            conn.close()
            raise ValueError("用户名已存在")

    def authenticate(self, username: str, password: str) -> Optional[Dict]:
        """用户认证"""
        conn = self.db_manager.get_mysql_connection()
        cursor = conn.cursor()

        cursor.execute('''
            SELECT id, username, password_hash, email, status
            FROM users
            WHERE username = %s
        ''', (username,))

        result = cursor.fetchone()
        conn.close()

        if not result:
            return None

        user_id, username, password_hash, email, status = result

        if status == 0:
            return None

        if check_password_hash(password_hash, password):
            return {
                'id': user_id,
                'username': username,
                'email': email,
                'status': status
            }

        return None

    def get_user_by_id(self, user_id: int) -> Optional[Dict]:
        """根据ID获取用户信息"""
        conn = self.db_manager.get_mysql_connection()
        cursor = conn.cursor()

        cursor.execute('''
            SELECT id, username, email, status
            FROM users
            WHERE id = %s
        ''', (user_id,))

        result = cursor.fetchone()
        conn.close()

        if result:
            return {
                'id': result[0],
                'username': result[1],
                'email': result[2],
                'status': result[3]
            }
        return None

    def assign_role(self, user_id: int, role_id: int):
        """为用户分配角色"""
        conn = self.db_manager.get_mysql_connection()
        cursor = conn.cursor()

        try:
            cursor.execute('''
                INSERT INTO user_roles (user_id, role_id)
                VALUES (%s, %s)
            ''', (user_id, role_id))
            conn.close()
        except pymysql.IntegrityError:
            conn.close()
            raise ValueError("用户角色关系已存在")

# 认证装饰器
def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        token = request.headers.get('Authorization')
        if not token:
            return jsonify({'error': '缺少认证token'}), 401

        if token.startswith('Bearer '):
            token = token[7:]

        token_data = token_manager.verify_token(token)
        if not token_data:
            return jsonify({'error': '无效或过期的token'}), 401

        user_id = token_data['user_id']
        user = user_manager.get_user_by_id(user_id)
        if not user or user['status'] == 0:
            return jsonify({'error': '用户不存在或已禁用'}), 401

        g.current_user = user
        return f(*args, **kwargs)

    return decorated_function

# 权限检查装饰器
def permission_required(resource: str, action: str):
    def decorator(f):
        @wraps(f)
        @login_required
        def decorated_function(*args, **kwargs):
            user_id = g.current_user['id']

            if not permission_manager.check_permission(user_id, resource, action):
                return jsonify({'error': '权限不足'}), 403

            return f(*args, **kwargs)
        return decorated_function
    return decorator

# API路由
@app.route('/api/register', methods=['POST'])
def register():
    """用户注册"""
    data = request.get_json()
    username = data.get('username')
    password = data.get('password')
    email = data.get('email')

    if not username or not password:
        return jsonify({'error': '用户名和密码不能为空'}), 400

    try:
        user = user_manager.create_user(username, password, email)
        return jsonify({
            'message': '注册成功',
            'user': user
        }), 201
    except ValueError as e:
        return jsonify({'error': str(e)}), 400

@app.route('/api/login', methods=['POST'])
def login():
    """用户登录"""
    data = request.get_json()
    username = data.get('username')
    password = data.get('password')

    if not username or not password:
        return jsonify({'error': '用户名和密码不能为空'}), 400

    user = user_manager.authenticate(username, password)
    if not user:
        return jsonify({'error': '用户名或密码错误'}), 401

    token = token_manager.generate_token(user['id'])

    return jsonify({
        'message': '登录成功',
        'token': token,
        'user': user
    })

@app.route('/api/logout', methods=['POST'])
@login_required
def logout():
    """用户登出"""
    token = request.headers.get('Authorization')
    if token and token.startswith('Bearer '):
        token = token[7:]
        token_manager.revoke_token(token)

    return jsonify({'message': '登出成功'})

@app.route('/api/profile', methods=['GET'])

def profile():

    permissions = list(permission_manager.get_user_permissions(user_id))

    return jsonify({
        'user': g.current_user,
        'permissions': permissions
    })

    # RBAC认证权限系统部署指南

## 系统概述

这是一个基于RBAC (Role-Based Access Control) 模型的动态认证和权限访问控制系统，具有以下特性：

- **Token认证**: 基于Redis的Token管理，支持过期和撤销
- **动态权限**: 从MySQL数据库动态加载权限信息
- **Redis缓存**: 用户权限信息缓存，提高性能
- **完整API**: 提供用户管理、角色管理、权限管理的完整API
- **装饰器权限控制**: 通过装饰器实现细粒度的权限控制

    权限系统说明

    权限格式

    权限采用 resource:action 格式，例如： - user:read - 用户读取权限 - user:create - 用户创建权限 - product:update - 产品更新权限

    权限检查流程

    1. 从请求头提取Token
    2. 验证Token有效性
    3. 获取用户ID
    4. 从Redis缓存获取用户权限（如果没有则从数据库查询）
    5. 检查用户是否具有所需权限
    6. 允许或拒绝访问

    缓存机制

    • Token缓存: 存储在Redis中，设置过期时间
    • 权限缓存: 用户权限信息缓存5分钟，提高性能
    • 缓存失效: 角色权限变更时自动清除相关用户的权限缓存

    def multiple_permissions_required(*permissions):
    """需要多个权限的装饰器"""
    def decorator(f):
        @wraps(f)
        @login_required
        def decorated_function(*args, **kwargs):
            user_id = g.current_user['id']

            for resource, action in permissions:
                if not permission_manager.check_permission(user_id, resource, action):
                    return jsonify({'error': f'缺少权限: {resource}:{action}'}), 403

            return f(*args, **kwargs)
        return decorated_function
    return decorator

    这个RBAC认证权限系统提供了完整的用户认证、角色管理和权限控制功能。通过Redis缓存和MySQL存储的组合，实现了高性能的动态权限检查。系统具有良好的扩展性，可以根据业务需求轻松添加新的权限控制逻辑。

    
我已经为您创建了一个完整的基于RBAC模型的认证和权限访问控制系统。这个系统包含以下核心功能：

## 🎯 核心特性

**1. 完整的RBAC模型**
- 用户(Users) → 角色(Roles) → 权限(Permissions)
- 支持多角色用户和动态权限分配

**2. Token认证系统**
- 基于Redis的Token管理
- 自动过期和撤销机制
- 支持Bearer Token格式

**3. 动态权限加载**
- 从MySQL数据库实时加载权限
- Redis缓存提升性能
- 权限变更时自动清除缓存

**4. 装饰器权限控制**
- `@login_required` - 登录验证
- `@permission_required(resource, action)` - 权限验证
- 支持细粒度权限控制
